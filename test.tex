\def\LTORIG#1{%
  \expandafter\newbox\csname lt#1\endcsname
  \global\expandafter\setbox\csname lt#1\endcsname
  \vtop{\BASEDIR\normalsize\adjustbaseline\noindent
 	\textbf{\kern5zw未読込}
	  \begin{description}
		\def\makelabel##1{\hbox to 4em{##1}}
	   \item[Top] \A{t}{}
	   \item[Center] \leavevmode\kern3zw\A{c}{}
	   \item[Math] $\A{c}{}$
	   \item[Bottom] \leavevmode\kern3zw\A{b}{}
	  \end{description}
  }}
